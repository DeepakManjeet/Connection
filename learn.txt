pipeline {
    agent any

    stages {
        stage('Call API') {
            steps {
                script {
                    def response = httpRequest(
                        url: 'https://reqres.in/api/users?page=2',
                        validResponseCodes: '200',
                                            )
                    echo "API response status: ${response.status}"
                    echo "API response body: ${response.content}"
                }
            }
        }
    }
}
